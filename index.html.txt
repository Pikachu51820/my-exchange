<script>
    const JSON_URL = 'https://tsunami.cash';
    const REF_LINK = 'https://tsunami.cash';

    async function loadRates() {
        try {
            const response = await fetch(JSON_URL);
            const data = await response.json();
            const tableBody = document.getElementById('tableBody');
            
            // Фильтруем популярные направления, чтобы не перегружать страницу
            const popular = ['BTC', 'ETH', 'USDT', 'RUB', 'SBERRUB', 'TCSBRUB'];
            
            tableBody.innerHTML = ''; // Очистка

            data.forEach(item => {
                if (popular.includes(item.from)) {
                    const row = `
                        <tr>
                            <td><b>${item.from}</b></td>
                            <td>${item.to}</td>
                            <td>1 ${item.from} = ${parseFloat(item.in).toFixed(2)} ${item.to}</td>
                            <td><a href="${REF_LINK}" class="btn-exchange">Обменять</a></td>
                        </tr>
                    `;
                    tableBody.innerHTML += row;
                }
            });
        } catch (error) {
            document.getElementById('tableBody').innerHTML = '<tr><td colspan="4">Ошибка загрузки курсов. Попробуйте обновить страницу.</td></tr>';
        }
    }

    loadRates();
</script>
